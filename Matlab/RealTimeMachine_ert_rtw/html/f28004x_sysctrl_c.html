   84   // InitSysCtrl - Initialization of system resources.
   85   //
   86   void
   87   InitSysCtrl(void)
   88   {
   89       //
   90       // Disable the watchdog
   91       //
   92       DisableDog();
   93   
   94   #ifdef _FLASH
   95       //
   96       // Copy time critical code and Flash setup code to RAM
   97       // This includes the following functions:  InitFlash();
   98       // The  RamfuncsLoadStart, RamfuncsLoadSize, and RamfuncsRunStart
   99       // symbols are created by the linker. Refer to the device .cmd file.
  100       //
  101       memcpy(&RamfuncsRunStart, &RamfuncsLoadStart, (size_t)&RamfuncsLoadSize);
  102   
  103       //
  104       // Call Flash Initialization to setup flash waitstates
  105       // This function must reside in RAM
  106       //
  107       InitFlash();
  108   #endif
  109   
  110       //
  111       // PLLSYSCLK = (XTAL_OSC) * (IMULT + FMULT) / (PLLSYSCLKDIV)
  112       //
  113       InitSysPll(XTAL_OSC,IMULT_10,FMULT_0,PLLCLK_BY_2);
  114   
  115   #ifndef _FLASH
  116       //
  117       // Call Device_cal function when run using debugger
  118       // This function is called as part of the Boot code. The function is called
  119       // in the InitSysCtrl function since during debug time resets, the boot code
  120       // will not be executed and the gel script will reinitialize all the
  121       // registers and the calibrated values will be lost.
  122       //
  123       Device_cal();
  124   #endif
  125   
  126       //
  127       // Turn on all peripherals
  128       //
  129       InitPeripheralClocks();
  130   }

  248   //
  249   // InitFlash - This function initializes the Flash Control registers
  250   //                  CAUTION
  251   // This function MUST be executed out of RAM. Executing it
  252   // out of OTP/Flash will yield unpredictable results
  253   //
  254   #ifdef __cplusplus
  255   #pragma CODE_SECTION(".TI.ramfunc");
  256   #endif
  257   void
  258   InitFlash(void)
  259   {
  260       EALLOW;
  261   
  262       //
  263       // At reset bank and pump are in sleep
  264       // A Flash access will power up the bank and pump automatically
  265       // After a Flash access, bank and pump go to low power mode (configurable
  266       // in FBFALLBACK/FPAC1 registers)- if there is no further access to flash
  267       // Power up Flash bank and pump and this also sets the fall back mode of
  268       // flash and pump as active
  269       //
  270       Flash0CtrlRegs.FPAC1.bit.PMPPWR = 0x1;
  271       Flash0CtrlRegs.FBFALLBACK.bit.BNKPWR0 = 0x3;
  272       Flash0CtrlRegs.FBFALLBACK.bit.BNKPWR1 = 0x3;
  273   
  274       //
  275       // Disable Cache and prefetch mechanism before changing wait states
  276       //
  277       Flash0CtrlRegs.FRD_INTF_CTRL.bit.DATA_CACHE_EN = 0;
  278       Flash0CtrlRegs.FRD_INTF_CTRL.bit.PREFETCH_EN = 0;
  279   
  280       //
  281       // Set waitstates according to frequency
  282       //                CAUTION
  283       // Minimum waitstates required for the flash operating
  284       // at a given CPU rate must be characterized by TI.
  285       // Refer to the datasheet for the latest information.
  286       //
  287   #if CPU_FRQ_100MHZ
  288       if((ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL == 0x0)   ||
  289          (ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL == 0x2)   ||
  290          (ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL == 0x3))
  291       {
  292           Flash0CtrlRegs.FRDCNTL.bit.RWAIT = 0x5;
  293       }
  294       else
  295       {
  296           Flash0CtrlRegs.FRDCNTL.bit.RWAIT = 0x4;
  297       }
  298   #endif
  299   
  300       //
  301       // Enable Cache and prefetch mechanism to improve performance
  302       // of code executed from Flash.
  303       //
  304       Flash0CtrlRegs.FRD_INTF_CTRL.bit.DATA_CACHE_EN = 1;
  305       Flash0CtrlRegs.FRD_INTF_CTRL.bit.PREFETCH_EN = 1;
  306   
  307       //
  308       // At reset, ECC is enabled. If it is disabled by application software
  309       // and if application again wants to enable ECC
  310       //
  311       Flash0EccRegs.ECC_ENABLE.bit.ENABLE = 0xA;
  312   
  313       EDIS;
  314   
  315       //
  316       // Force a pipeline flush to ensure that the write to
  317       // the last register configured occurs before returning.
  318       //
  319       __asm(" RPT #7 || NOP");
  320   }

  322   //
  323   // FlashOff - This function powers down the flash
  324   //                   CAUTION
  325   // This function MUST be executed out of RAM. Executing it
  326   // out of OTP/Flash will yield unpredictable results.
  327   // Note: a flash access after the flash pump and banks are powered down will
  328   // wake the pump and bank
  329   //
  330   #ifdef __cplusplus
  331   #pragma CODE_SECTION(".TI.ramfunc");
  332   #endif
  333   void
  334   FlashOff(void)
  335   {
  336       EALLOW;
  337   
  338       //
  339       // Configure the fallback power mode as sleep
  340       //
  341       Flash0CtrlRegs.FBFALLBACK.bit.BNKPWR0 = 0;
  342       Flash0CtrlRegs.FBFALLBACK.bit.BNKPWR1 = 0;
  343   
  344       //
  345       // Configure the fallback power mode as sleep
  346       //
  347       Flash0CtrlRegs.FPAC1.bit.PMPPWR = 0;
  348   
  349       EDIS;
  350   }

  369   DisableDog(void)

  371       volatile Uint16 temp;
  372       EALLOW;

  374       //
  375       // Grab the clock config so we don't clobber it

  382   //
  383   // InitPll - This function initializes the PLL registers.
  384   //
  385   // Note: This function uses the DCC to check that the PLLRAWCLK is running at
  386   // the expected rate. If you are using the DCC, you must back up its
  387   // configuration before calling this function and restore it afterward.
  388   //
  389   void
  390   InitSysPll(Uint16 clock_source, Uint16 imult, Uint16 fmult, Uint16 divsel)
  391   {
  392       Uint32 timeout, retries, temp_syspllmult, pllLockStatus;
  393       bool status;
  394   
  395       if(((clock_source & 0x3) == ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL)    &&
  396          (((clock_source & 0x4) >> 2) == ClkCfgRegs.XTALCR.bit.SE)           &&
  397                        (imult  == ClkCfgRegs.SYSPLLMULT.bit.IMULT)           &&
  398                        (fmult  == ClkCfgRegs.SYSPLLMULT.bit.FMULT)           &&
  399                        (divsel == ClkCfgRegs.SYSCLKDIVSEL.bit.PLLSYSCLKDIV))
  400       {
  401           //
  402           // Everything is set as required, so just return
  403           //
  404           return;
  405       }
  406   
  407       if(((clock_source & 0x3) != ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL) ||
  408          (((clock_source & 0x4) >> 2) != ClkCfgRegs.XTALCR.bit.SE))
  409       {
  410           switch (clock_source)
  411           {
  412               case INT_OSC1:
  413                   SysIntOsc1Sel();
  414                   break;
  415   
  416               case INT_OSC2:
  417                   SysIntOsc2Sel();
  418                   break;
  419   
  420               case XTAL_OSC:
  421                   SysXtalOscSel();
  422                   break;
  423   
  424               case XTAL_OSC_SE:
  425                   SysXtalOscSESel();
  426                   break;
  427           }
  428       }
  429   
  430       EALLOW;
  431   
  432       //
  433       // First modify the PLL multipliers
  434       //
  435       if(imult != ClkCfgRegs.SYSPLLMULT.bit.IMULT ||
  436          fmult != ClkCfgRegs.SYSPLLMULT.bit.FMULT)
  437       {
  438           //
  439           // Bypass PLL and set dividers to /1
  440           //
  441           ClkCfgRegs.SYSPLLCTL1.bit.PLLCLKEN = 0;
  442   
  443           //
  444           // Delay of at least 60 OSCCLK cycles required post PLL bypass
  445           //
  446           asm(" RPT #60 || NOP");
  447   
  448           ClkCfgRegs.SYSCLKDIVSEL.bit.PLLSYSCLKDIV = 0;
  449   
  450           //
  451           // Evaluate PLL multipliers
  452           //
  453           temp_syspllmult = ((fmult << 8U) | imult);
  454   
  455           //
  456           // Loop to retry locking the PLL should the DCC module indicate
  457           // that it was not successful.
  458           //
  459           for(retries = 0; (retries < PLL_RETRIES); retries++)
  460           {
  461               //
  462               // Disable SYSPLL (Turn off PLL)
  463               //
  464               ClkCfgRegs.SYSPLLCTL1.bit.PLLEN = 0;
  465   
  466               //
  467               // Atleast 33 cycle delay between powerdown to powerup
  468               //
  469               asm(" RPT #33 || NOP");
  470   
  471               //
  472               // Program PLL multipliers
  473               //
  474               ClkCfgRegs.SYSPLLMULT.all = temp_syspllmult;
  475   
  476               //
  477               // Enable SYSPLL
  478               //
  479               ClkCfgRegs.SYSPLLCTL1.bit.PLLEN = 1;
  480   
  481               timeout = PLL_LOCK_TIMEOUT;
  482               pllLockStatus = ClkCfgRegs.SYSPLLSTS.bit.LOCKS;
  483   
  484               //
  485               // Wait for the SYSPLL lock
  486               //
  487               while((pllLockStatus != 1) && (timeout != 0U))
  488               {
  489                   pllLockStatus = ClkCfgRegs.SYSPLLSTS.bit.LOCKS;
  490                   timeout--;
  491               }
  492   
  493               EDIS;
  494   
  495               status = IsPLLValid(clock_source, imult, fmult);
  496   
  497               //
  498               // Check DCC Status, if no error break the loop
  499               //
  500               if(status)
  501               {
  502                   break;
  503               }
  504           }
  505       }
  506       else
  507       {
  508           status = true;
  509       }
  510   
  511       if(status)
  512       {
  513           EALLOW;
  514           //
  515           // Set divider to produce slower output frequency to limit current increase
  516           //
  517           if(divsel != PLLCLK_BY_126)
  518           {
  519               ClkCfgRegs.SYSCLKDIVSEL.bit.PLLSYSCLKDIV = divsel + 1;
  520           }
  521           else
  522           {
  523               ClkCfgRegs.SYSCLKDIVSEL.bit.PLLSYSCLKDIV = divsel;
  524           }
  525   
  526           //
  527           // Enable PLLSYSCLK is fed from system PLL clock
  528           //
  529           ClkCfgRegs.SYSPLLCTL1.bit.PLLCLKEN = 1;
  530   
  531           //
  532           // Small 100 cycle delay
  533           //
  534           asm(" RPT #100 || NOP");
  535   
  536           //
  537           // Set the divider to user value
  538           //
  539           ClkCfgRegs.SYSCLKDIVSEL.bit.PLLSYSCLKDIV = divsel;
  540           EDIS;
  541       }
  542   }

  544   //
  545   // CsmUnlock - This function unlocks the CSM. User must replace 0xFFFF's with
  546   // current password for the DSP.
  547   //
  548   Uint16
  549   CsmUnlock()
  550   {
  551       //
  552       // Write to the key registers to unlock the device. The 0x0FFFFFFFF's
  553       // are dummmy passwords. User should replace them with the correct password
  554       // for the DSP.
  555       // Note: F28004x has default password keys which are not all 0xFFFFFFFF.
  556       // See DCSM chapter of Technical Reference Manual for default passwords.
  557       EALLOW;
  558   
  559       DcsmBank0Z1Regs.Z1_CSMKEY0 = 0xFFFFFFFF;
  560       DcsmBank0Z1Regs.Z1_CSMKEY1 = 0xFFFFFFFF;
  561       DcsmBank0Z1Regs.Z1_CSMKEY2 = 0xFFFFFFFF;
  562       DcsmBank0Z1Regs.Z1_CSMKEY3 = 0xFFFFFFFF;
  563   
  564       DcsmBank0Z2Regs.Z2_CSMKEY0 = 0xFFFFFFFF;
  565       DcsmBank0Z2Regs.Z2_CSMKEY1 = 0xFFFFFFFF;
  566       DcsmBank0Z2Regs.Z2_CSMKEY2 = 0xFFFFFFFF;
  567       DcsmBank0Z2Regs.Z2_CSMKEY3 = 0xFFFFFFFF;
  568       EDIS;
  569   
  570       return 0;
  571   }

  574   // SysIntOsc1Sel - This function switches to Internal Oscillator 1 and turns
  575   // off all other clock sources to minimize power consumption
  576   //
  577   void
  578   SysIntOsc1Sel (void)
  579   {
  580       EALLOW;
  581       ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL = 2; // Clk Src = INTOSC1
  582       ClkCfgRegs.XTALCR.bit.OSCOFF=1;             // Turn off XTALOSC
  583       EDIS;
  584   }

  586   //
  587   // SysIntOsc2Sel - This function switches to Internal oscillator 2 from
  588   // External Oscillator and turns off all other clock sources to minimize
  589   // power consumption
  590   // NOTE: If there is no external clock connection, when switching from
  591   //       INTOSC1 to INTOSC2, EXTOSC and XLCKIN must be turned OFF prior
  592   //       to switching to internal oscillator 1
  593   //
  594   void
  595   SysIntOsc2Sel (void)
  596   {
  597       EALLOW;
  598       ClkCfgRegs.CLKSRCCTL1.bit.INTOSC2OFF=0;         // Turn on INTOSC2
  599       SYSCTRL_CLKSRCCTL1_DELAY;
  600       ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL = 0;     // Clk Src = INTOSC2
  601       ClkCfgRegs.XTALCR.bit.OSCOFF=1;                 // Turn off XTALOSC
  602       EDIS;
  603   }

  605   //
  606   // PollX1Counter - Clear the X1CNT counter and then wait for it to saturate
  607   // four times.
  608   //
  609   static void
  610   PollX1Counter(void)
  611   {
  612       Uint16 loopCount = 0;
  613   
  614       //
  615       // Delay for 1 ms while the XTAL powers up
  616       //
  617       // 2000 loops, 5 cycles per loop + 9 cycles overhead = 10009 cycles
  618       //
  619       F28x_usDelay(2000);
  620   
  621       //
  622       // Clear and saturate X1CNT 4 times to guarantee operation
  623       //
  624       do
  625       {
  626           //
  627           // Keep clearing the counter until it is no longer saturated
  628           //
  629           while(ClkCfgRegs.X1CNT.all > 0x1FF)
  630           {
  631               ClkCfgRegs.X1CNT.bit.CLR = 1;
  632           }
  633   
  634           //
  635           // Wait for the X1 clock to saturate
  636           //
  637           while(ClkCfgRegs.X1CNT.all != 0x3FFU)
  638           {
  639               ;
  640           }
  641   
  642           //
  643           // Increment the counter
  644           //
  645           loopCount++;
  646       }while(loopCount < 4);
  647   }

  650   // SysXtalOscSel - This function switches to External CRYSTAL oscillator and
  651   // turns off all other clock sources to minimize power consumption. This option
  652   // may not be available on all device packages
  653   //
  654   void
  655   SysXtalOscSel (void)
  656   {
  657       EALLOW;
  658       ClkCfgRegs.XTALCR.bit.OSCOFF = 0;     // Turn on XTALOSC
  659       ClkCfgRegs.XTALCR.bit.SE = 0;         // Select crystal mode
  660       EDIS;
  661   
  662       //
  663       // Wait for the X1 clock to saturate
  664       //
  665       PollX1Counter();
  666   
  667       //
  668       // Select XTAL as the oscillator source
  669       //
  670       EALLOW;
  671       ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL = 1;
  672       EDIS;
  673   
  674       //
  675       // If a missing clock failure was detected, try waiting for the X1 counter
  676       // to saturate again. Consider modifying this code to add a 10ms timeout.
  677       //
  678       while(ClkCfgRegs.MCDCR.bit.MCLKSTS != 0)
  679       {
  680           EALLOW;
  681           ClkCfgRegs.MCDCR.bit.MCLKCLR = 1;
  682           EDIS;
  683   
  684           //
  685           // Wait for the X1 clock to saturate
  686           //
  687           PollX1Counter();
  688   
  689           //
  690           // Select XTAL as the oscillator source
  691           //
  692           EALLOW;
  693           ClkCfgRegs.CLKSRCCTL1.bit.OSCCLKSRCSEL = 1;
  694           EDIS;
  695       }
  696   }

  759   //*****************************************************************************
  760   //
  761   // SysCtl_isPLLValid()
  762   //
  763   //*****************************************************************************
  764   bool
  765   IsPLLValid(Uint16 oscSource, Uint16 imult, Uint16 fmult)
  766   {
  767       Uint32 dccCounterSeed0, dccCounterSeed1, dccValidSeed0;
  768   
  769       //
  770       // Setting Counter0 & Valid Seed Value with +/-2% tolerance
  771       //
  772       dccCounterSeed0 = DCC_COUNTER0_WINDOW - 2U;
  773       dccValidSeed0 = 4U;
  774   
  775       //
  776       // Multiplying Counter-0 window with PLL Integer Multiplier
  777       //
  778       dccCounterSeed1 = DCC_COUNTER0_WINDOW * imult;
  779   
  780       //
  781       // Multiplying Counter-0 window with PLL Fractional Multiplier
  782       //
  783       switch(fmult)
  784       {
  785           case FMULT_0pt25:
  786               //
  787               // FMULT * CNTR0 Window = 0.25 * 100 = 25, gets added to cntr0
  788               // seed value
  789               //
  790               dccCounterSeed1 = dccCounterSeed1 + 25U;
  791               break;
  792           case FMULT_0pt5:
  793               //
  794               // FMULT * CNTR0 Window = 0.5 * 100 = 50, gets added to cntr0
  795               // seed value
  796               //
  797               dccCounterSeed1 = dccCounterSeed1 + 50U;
  798               break;
  799           case FMULT_0pt75:
  800               //
  801               // FMULT * CNTR0 Window = 0.75 * 100 = 75, gets added to cntr0
  802               // seed value
  803               //
  804               dccCounterSeed1 = dccCounterSeed1 + 75U;
  805               break;
  806           default:
  807               //
  808               // No fractional multiplier
  809               //
  810               dccCounterSeed1 = dccCounterSeed1;
  811               break;
  812       }
  813   
  814       //
  815       // Enable Peripheral Clock Domain PCLKCR21 for DCC
  816       //
  817       EALLOW;
  818       CpuSysRegs.PCLKCR21.bit.DCC_0 = 1;
  819   
  820       //
  821       // Clear Error & Done Flag
  822       //
  823       Dcc0Regs.DCCSTATUS.bit.ERR = 1;
  824       Dcc0Regs.DCCSTATUS.bit.DONE = 1;
  825   
  826       //
  827       // Disable DCC
  828       //
  829       Dcc0Regs.DCCGCTRL.bit.DCCENA = 0x5;
  830   
  831       //
  832       // Disable Error Signal
  833       //
  834       Dcc0Regs.DCCGCTRL.bit.ERRENA = 0x5;
  835   
  836       //
  837       // Disable Done Signal
  838       //
  839       Dcc0Regs.DCCGCTRL.bit.DONEENA = 0x5;
  840   
  841       //
  842       // Configure Clock Source0 to whatever is set as a clock source for PLL
  843       //
  844       switch(oscSource)
  845       {
  846           case INT_OSC1:
  847               Dcc0Regs.DCCCLKSRC0.bit.CLKSRC0 = 1; // Clk Src0 = INTOSC1
  848               break;
  849   
  850           case INT_OSC2:
  851               Dcc0Regs.DCCCLKSRC0.bit.CLKSRC0 = 2; // Clk Src0 = INTOSC2
  852               break;
  853   
  854           case XTAL_OSC:
  855           case XTAL_OSC_SE:
  856               Dcc0Regs.DCCCLKSRC0.bit.CLKSRC0 = 0; // Clk Src0 = XTAL
  857               break;
  858       }
  859   
  860       //
  861       // Configure Clock Source1 to PLL
  862       //
  863       Dcc0Regs.DCCCLKSRC1.bit.KEY = 0xA; // Clk Src1 Key to enable clock source selection for count1
  864       Dcc0Regs.DCCCLKSRC1.bit.CLKSRC1 = 0; // Clk Src1 = PLL
  865   
  866       //
  867       // Configure COUNTER-0, COUNTER-1 & Valid Window
  868       //
  869       Dcc0Regs.DCCCNTSEED0.bit.COUNTSEED0 = dccCounterSeed0; // Loaded Counter0 Value
  870       Dcc0Regs.DCCVALIDSEED0.bit.VALIDSEED = dccValidSeed0;  // Loaded Valid Value
  871       Dcc0Regs.DCCCNTSEED1.bit.COUNTSEED1 = dccCounterSeed1; // Loaded Counter1 Value
  872   
  873       //
  874       // Enable Single Shot Mode
  875       //
  876       Dcc0Regs.DCCGCTRL.bit.SINGLESHOT = 0xA;
  877   
  878       //
  879       // Enable Error Signal
  880       //
  881       Dcc0Regs.DCCGCTRL.bit.ERRENA = 0xA;
  882   
  883       //
  884       // Enable Done Signal
  885       //
  886       Dcc0Regs.DCCGCTRL.bit.DONEENA = 0xA;
  887   
  888       //
  889       // Enable DCC to start counting
  890       //
  891       Dcc0Regs.DCCGCTRL.bit.DCCENA = 0xA;
  892       EDIS;
  893   
  894       //
  895       // Wait until Error or Done Flag is generated
  896       //
  897       while((Dcc0Regs.DCCSTATUS.all & 3) == 0)
  898       {
  899       }
  900   
  901       //
  902       // Returns true if DCC completes without error
  903       //
  904       return((Dcc0Regs.DCCSTATUS.all & 3) == 2);
  905   
  906   }
