   53   //
   54   // InitGpio - Sets all pins to be muxed to GPIO in input mode with pull-ups
   55   // enabled.Also resets CPU control to CPU1 and disables open drain
   56   // and polarity inversion and sets the qualification to synchronous.
   57   // Also unlocks all GPIOs. Only one CPU should call this function.
   58   //
   59   void
   60   InitGpio()
   61   {
   62       volatile Uint32 *gpioBaseAddr;
   63       Uint16 regOffset;
   64   
   65       //
   66       // Disable pin locks
   67       //
   68       EALLOW;
   69       GpioCtrlRegs.GPALOCK.all = 0x00000000;
   70       GpioCtrlRegs.GPBLOCK.all = 0x00000000;
   71       GpioCtrlRegs.GPHLOCK.all = 0x00000000;
   72   
   73       //
   74       // Fill all registers with zeros. Writing to each register separately
   75       // for three GPIO modules would make this function *very* long.
   76       // Fortunately, we'd be writing them all with zeros anyway,
   77       // so this saves a lot of space.
   78       //
   79       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;
   80       for (regOffset = 0; regOffset < sizeof(GpioCtrlRegs)/2; regOffset++)
   81       {
   82           //
   83           // Must avoid enabling pull-ups on all pins. GPyPUD is offset
   84           // 0x0C in each register group of 0x40 words. Since this is a
   85           // 32-bit pointer, the addresses must be divided by 2.
   86           //
   87           // Also, to avoid changing pin muxing of the emulator pins to regular
   88           // GPIOs, skip GPBMUX1 (0x46) and GPBGMUX1 (0x60).
   89           //
   90           if ((regOffset % (0x40/2) != (0x0C/2)) && (regOffset != (0x46/2)) &&
   91               (regOffset != (0x60/2)))
   92           {
   93               gpioBaseAddr[regOffset] = 0x00000000;
   94           }
   95       }
   96   
   97       gpioBaseAddr = (Uint32 *)&GpioDataRegs;
   98       for (regOffset = 0; regOffset < sizeof(GpioDataRegs)/2; regOffset++)
   99       {
  100           gpioBaseAddr[regOffset] = 0x00000000;
  101       }
  102   
  103       EDIS;
  104   }

  106   //
  107   // GPIO_SetupPinMux - Set the peripheral muxing for the specified pin.
  108   // The appropriate parameters can be found in the pinout spreadsheet.
  109   //
  110   void
  111   GPIO_SetupPinMux(Uint16 gpioNumber, Uint16 cpu, Uint16 muxPosition)
  112   {
  113       volatile Uint32 *gpioBaseAddr;
  114       volatile Uint32 *mux, *gmux, *csel;
  115       Uint16 pin32, pin16, pin8;
  116   
  117       pin32 = gpioNumber % 32;
  118       pin16 = gpioNumber % 16;
  119       pin8 = gpioNumber % 8;
  120       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (gpioNumber/32)*GPY_CTRL_OFFSET;
  121   
  122       //
  123       // Sanity check for valid cpu and peripheral values
  124       //
  125       if (cpu > GPIO_MUX_CPU1CLA || muxPosition > 0xF)
  126       {
  127           return;
  128       }
  129   
  130       //
  131       // Create pointers to the appropriate registers. This is a workaround
  132       // for the way GPIO registers are defined. The standard definition
  133       // in the header file makes it very easy to do named accesses of one
  134       // register or bit, but hard to do arbitrary numerical accesses. It's
  135       // easier to have an array of GPIO modules with identical registers,
  136       // including arrays for multi-register groups like GPyCSEL1-4. But
  137       // the header file doesn't define anything we can turn into an array,
  138       // so manual pointer arithmetic is used instead.
  139       //
  140       mux = gpioBaseAddr + GPYMUX + pin32/16;
  141       gmux = gpioBaseAddr + GPYGMUX + pin32/16;
  142       csel = gpioBaseAddr + GPYCSEL + pin32/8;
  143   
  144       //
  145       // Now for the actual function
  146       //
  147       EALLOW;
  148   
  149       //
  150       // To change the muxing, set the peripheral mux to 0/GPIO first to avoid
  151       // glitches, then change the group mux, then set the peripheral mux to
  152       // its target value. Finally, set the CPU select. This procedure is
  153       // described in the TRM. Unfortunately, since we don't know the pin in
  154       // advance we can't hardcode a bitfield reference, so there's some tricky
  155       // bit twiddling here.
  156       //
  157       *mux &= ~(0x3UL << (2*pin16));
  158       *gmux &= ~(0x3UL << (2*pin16));
  159       *gmux |= (Uint32)((muxPosition >> 2) & 0x3UL) << (2*pin16);
  160       *mux |= (Uint32)(muxPosition & 0x3UL) << (2*pin16);
  161   
  162       *csel &= ~(0x3L << (4*pin8));
  163       *csel |= (Uint32)(cpu & 0x3L) << (4*pin8);
  164   
  165       //
  166       // WARNING: This code does not touch the analog mode select registers.
  167       //
  168   
  169       EDIS;
  170   }

  172   //
  173   // GPIO_SetupPinOptions - Setup up the GPIO input/output options for the
  174   // specified pin. The flags are a 16-bit mask produced by ORing together
  175   // options. For input pins, the valid flags are:
  176   // GPIO_PULLUP  Enable pull-up
  177   // GPIO_INVERT  Enable input polarity inversion
  178   // GPIO_SYNC  Synchronize the input latch to PLLSYSCLK
  179   //              (default -- you don't need to specify this)
  180   // GPIO_QUAL3  Use 3-sample qualification
  181   // GPIO_QUAL6  Use 6-sample qualification
  182   // GPIO_ASYNC  Do not use synchronization or qualification
  183   // (Note: only one of SYNC, QUAL3, QUAL6, or ASYNC is allowed)
  184   //
  185   // For output pins, the valid flags are:
  186   // GPIO_OPENDRAIN  Output in open drain mode
  187   // GPIO_PULLUP    If open drain enabled, also enable the pull-up
  188   // and the input qualification flags (SYNC/QUAL3/QUAL6/SYNC) listed above.
  189   //
  190   // With no flags, the default input state is synchronous with no
  191   // pull-up or polarity inversion. The default output state is
  192   // the standard digital output.
  193   //
  194   void
  195   GPIO_SetupPinOptions(Uint16 gpioNumber, Uint16 output, Uint16 flags)
  196   {
  197       volatile Uint32 *gpioBaseAddr;
  198       volatile Uint32 *dir, *pud, *inv, *odr, *qsel;
  199       Uint32 pin32, pin16, pinMask, qual;
  200   
  201       pin32 = gpioNumber % 32;
  202       pin16 = gpioNumber % 16;
  203       pinMask = 1UL << pin32;
  204       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (gpioNumber/32)*GPY_CTRL_OFFSET;
  205   
  206       //
  207       // Create pointers to the appropriate registers. This is a workaround
  208       // for the way GPIO registers are defined. The standard definition
  209       // in the header file makes it very easy to do named accesses of one
  210       // register or bit, but hard to do arbitrary numerical accesses. It's
  211       // easier to have an array of GPIO modules with identical registers,
  212       // including arrays for multi-register groups like GPyQSEL1-2. But
  213       // the header file doesn't define anything we can turn into an array,
  214       // so manual pointer arithmetic is used instead.
  215       //
  216       dir = gpioBaseAddr + GPYDIR;
  217       pud = gpioBaseAddr + GPYPUD;
  218       inv = gpioBaseAddr + GPYINV;
  219       odr = gpioBaseAddr + GPYODR;
  220       qsel = gpioBaseAddr + GPYQSEL + pin32/16;
  221   
  222       EALLOW;
  223   
  224       //
  225       // Set the data direction
  226       //
  227       *dir &= ~pinMask;
  228       if (output == 1)
  229       {
  230           //
  231           // Output, with optional open drain mode and pull-up
  232           //
  233           *dir |= pinMask;
  234   
  235           //
  236           // Enable open drain if necessary
  237           //
  238           if (flags & GPIO_OPENDRAIN)
  239           {
  240               *odr |= pinMask;
  241           }
  242           else
  243           {
  244               *odr &= ~pinMask;
  245           }
  246   
  247           //
  248           // Enable pull-up if necessary. Open drain mode must be active.
  249           //
  250           if (flags & (GPIO_OPENDRAIN | GPIO_PULLUP))
  251           {
  252               *pud &= ~pinMask;
  253           }
  254           else
  255           {
  256               *pud |= pinMask;
  257           }
  258       }
  259   
  260       else
  261       {
  262           //
  263           // Input, with optional pull-up, qualification, and polarity inversion
  264           //
  265           *dir &= ~pinMask;
  266   
  267           //
  268           // Enable pull-up if necessary
  269           //
  270           if (flags & GPIO_PULLUP)
  271           {
  272               *pud &= ~pinMask;
  273           }
  274           else
  275           {
  276               *pud |= pinMask;
  277           }
  278   
  279           //
  280           // Invert polarity if necessary
  281           //
  282           if (flags & GPIO_INVERT)
  283           {
  284               *inv |= pinMask;
  285           }
  286           else
  287           {
  288               *inv &= ~pinMask;
  289           }
  290       }
  291   
  292       //
  293       // Extract the qualification parameter and load it into the register.
  294       // This is also needed for open drain outputs, so we might as well do it
  295       // all the time.
  296       //
  297       qual = (flags & GPIO_ASYNC) / GPIO_QUAL3;
  298       *qsel &= ~(0x3L << (2 * pin16));
  299       if (qual != 0x0)
  300       {
  301           *qsel |= qual << (2 * pin16);
  302       }
  303   
  304       EDIS;
  305   }

  307   //
  308   // GPIO_SetupLock - Enable or disable the GPIO register bit lock for the
  309   // specified pin. The valid flags are:
  310   // GPIO_UNLOCK  Unlock the pin setup register bits for the specified pin
  311   // GPIO_LOCK   Lock the pin setup register bits for the specified pin
  312   //
  313   void
  314   GPIO_SetupLock(Uint16 gpioNumber, Uint16 flags)
  315   {
  316       volatile Uint32 *gpioBaseAddr;
  317       volatile Uint32 *lock;
  318       Uint32 pin32, pinMask;
  319   
  320       pin32 = gpioNumber % 32;
  321       pinMask = 1UL << pin32;
  322       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (gpioNumber/32)*GPY_CTRL_OFFSET;
  323   
  324       //
  325       // Create pointers to the appropriate registers. This is a workaround
  326       // for the way GPIO registers are defined. The standard definition
  327       // in the header file makes it very easy to do named accesses of one
  328       // register or bit, but hard to do arbitrary numerical accesses. It's
  329       // easier to have an array of GPIO modules with identical registers,
  330       // including arrays for multi-register groups like GPyQSEL1-2. But
  331       // the header file doesn't define anything we can turn into an array,
  332       // so manual pointer arithmetic is used instead.
  333       //
  334       lock = gpioBaseAddr + GPYLOCK;
  335   
  336       EALLOW;
  337   
  338       if(flags)
  339       {
  340           //
  341           // Lock the pin
  342           //
  343           *lock |= pinMask;
  344       }
  345   
  346       else
  347       {
  348           //
  349           // Unlock the pin
  350           //
  351           *lock &= ~pinMask;
  352       }
  353       EDIS;
  354   }

  398   //
  399   // GPIO_ReadPin - Read the GPyDAT register bit for the specified pin.
  400   // Note that this returns the actual state of the pin, not the state of the
  401   // output latch.
  402   //
  403   Uint16
  404   GPIO_ReadPin(Uint16 gpioNumber)
  405   {
  406       volatile Uint32 *gpioDataReg;
  407       Uint16 pinVal;
  408   
  409       gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (gpioNumber/32)*GPY_DATA_OFFSET;
  410       pinVal = (gpioDataReg[GPYDAT] >> (gpioNumber % 32)) & 0x1;
  411   
  412       return pinVal;
  413   }
  414   
  415   //
  416   // GPIO_WritePin - Set the GPyDAT register bit for the specified pin.
  417   //
  418   void
  419   GPIO_WritePin(Uint16 gpioNumber, Uint16 outVal)
  420   {
  421       volatile Uint32 *gpioDataReg;
  422       Uint32 pinMask;
  423   
  424       gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (gpioNumber/32)*GPY_DATA_OFFSET;
  425       pinMask = 1UL << (gpioNumber % 32);
  426   
  427       if (outVal == 0)
  428       {
  429           gpioDataReg[GPYCLEAR] = pinMask;
  430       }
  431   
  432       else
  433       {
  434           gpioDataReg[GPYSET] = pinMask;
  435       }
  436   }
