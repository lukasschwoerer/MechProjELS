   48   /*Sets all pins to be muxed to GPIO in input mode with pull-ups enabled.
   49    * Also sets the qualification to synchronous.*/
   50   void InitGpio()
   51   {
   52       volatile Uint32 *gpioBaseAddr;
   53       Uint16 regOffset;
   54       EALLOW;
   55       /*Fill all registers with zeros. Writing to each register separately
   56        * for all GPIO modules would make this function *very* long. Fortunately,
   57        * we'd be writing them all with zeros anyway, so this saves a lot of space.*/
   58       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;
   59       for (regOffset = 0; regOffset < sizeof(GpioCtrlRegs)/2; regOffset++)
   60       {
   61           /*Hack to avoid enabling pull-ups on all pins. GPyPUD is offset
   62            * 0x0C in each register group of 0x40 words. Since this is a
   63            * 32-bit pointer, the addresses must be divided by 2.*/
   64           if (regOffset % (0x10/2) != (0x0C/2))
   65               gpioBaseAddr[regOffset] = 0x00000000;
   66       }
   67       
   68       gpioBaseAddr = (Uint32 *)&GpioDataRegs;
   69       for (regOffset = 0; regOffset < sizeof(GpioDataRegs)/2; regOffset++)
   70       {
   71           gpioBaseAddr[regOffset] = 0x00000000;
   72       }
   73       
   74       /* For concerto additional registers needs to be initialized*/
   75       #ifdef MW_PIL_TIC2000_CONCERTO
   76       /*Fill all registers with zeros. Writing to each register separately
   77        * for all GPIO modules would make this function *very* long. Fortunately,
   78        * we'd be writing them all with zeros anyway, so this saves a lot of space.*/
   79       gpioBaseAddr = (Uint32 *)&GpioG1CtrlRegs;
   80       for (regOffset = 0; regOffset < sizeof(GpioG1CtrlRegs)/2; regOffset++)
   81       {
   82           /*Hack to avoid enabling pull-ups on all pins. GPyPUD is offset
   83            * 0x0C in each register group of 0x40 words. Since this is a
   84            * 32-bit pointer, the addresses must be divided by 2.*/
   85           if (regOffset % (0x10/2) != (0x0C/2))
   86               gpioBaseAddr[regOffset] = 0x00000000;
   87       }
   88       
   89       gpioBaseAddr = (Uint32 *)&GpioG1DataRegs;
   90       for (regOffset = 0; regOffset < sizeof(GpioG1DataRegs)/2; regOffset++)
   91       {
   92           gpioBaseAddr[regOffset] = 0x00000000;
   93       }
   94       #endif //End of MW_PIL_TIC2000_CONCERTO
   95       EDIS;
   96   }
   97   #endif // End of MW_F2804X

   99   /*Set the peripheral muxing for the specified pin.
  100    * The appropriate parameters can be found in the GPIO Muxed Pins table.
  101    * Use the GPIO index row (0 to 4) to select a muxing option for the GPIO.*/
  102   void GPIO_SetupPinMux(Uint16 pin, Uint16 cpu, Uint16 peripheral)
  103   {
  104       volatile Uint32 *gpioBaseAddr;
  105       volatile Uint32 *mux;
  106       Uint16 pin32, pin16;
  107       
  108       pin32 = pin % 32;
  109       pin16 = pin % 16;
  110       #ifdef MW_PIL_TIC2000_CONCERTO
  111       if (pin >= 128)
  112       {
  113           #ifdef F28M36x_DEVICE_H
  114           if (pin <= 135)
  115           {
  116               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;
  117           }
  118           else if (pin >=192 && pin <= 199)
  119           {
  120               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x1080/2);
  121           }
  122           #endif //End of F28M36x_DEVICE_H
  123           #ifdef F28M35x_DEVICE_H
  124           if (pin <= 135)
  125           {
  126               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x1000/2);
  127           }
  128           #endif //End of F28M35x_DEVICE_H
  129       }
  130       else
  131       {
  132           #ifdef F28M36x_DEVICE_H
  133           gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x80/2) + (pin/32)*GPY_CTRL_OFFSET;
  134           #endif //End of F28M36x_DEVICE_H
  135           #ifdef F28M35x_DEVICE_H
  136           gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;
  137           #endif //End of F28M35x_DEVICE_H
  138       }
  139       #else
  140       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;
  141       #endif // End of MW_PIL_TIC2000_CONCERTO
  142       
  143       /*Sanity check for valid peripheral values*/
  144       if (peripheral > 0x4)
  145           return;
  146       
  147       /*Create pointers to the appropriate registers. This is a workaround
  148        * for the way GPIO registers are defined. The standard definition
  149        * in the header file makes it very easy to do named accesses of one
  150        * register or bit, but hard to do arbitrary numerical accesses. It's
  151        * easier to have an array of GPIO modules with identical registers,
  152        * including arrays for multi-register groups like GPyCSEL1-4. But
  153        * the header file doesn't define anything we can turn into an array,
  154        * so manual pointer arithmetic is used instead.*/
  155       mux = gpioBaseAddr + GPYMUX + pin32/16;
  156       
  157       //Now for the actual function
  158       EALLOW;
  159       
  160       /* Unfortunately, since we don't know the pin in
  161        * advance we can't hardcode a bitfield reference, so there's some tricky
  162        * bit twiddling here.*/
  163       *mux &= ~(0x3UL << (2*pin16));
  164       *mux |= (Uint32)(peripheral & 0x3UL) << (2*pin16);
  165       
  166       //WARNING: This code does not touch the analog mode select registers,
  167       //which are needed to give the USB module control of its IOs.
  168       EDIS;
  169   }

  171   /*Setup up the GPIO input/output options for the specified pin.
  172    * The flags are a 16-bit mask produced by ORing together options.
  173    * For input pins, the valid flags are:
  174    * GPIO_PULLUP    Enable pull-up
  175    * GPIO_SYNC        Synchronize the input latch to PLLSYSCLK (default -- you don't need to specify this)
  176    * GPIO_QUAL3    Use 3-sample qualification
  177    * GPIO_QUAL6    Use 6-sample qualification
  178    * GPIO_ASYNC    Do not use synchronization or qualification
  179    * (Note: only one of SYNC, QUAL3, QUAL6, or ASYNC is allowed)
  180    * For output pins, the valid flags are:
  181    * GPIO_PULLUP        If open drain enabled, also enable the pull-up
  182    * and the input qualification flags (SYNC/QUAL3/QUAL6/SYNC) listed above.
  183    * With no flags, the default input state is synchronous with no pull-up or polarity inversion.
  184    * The default output state is the standard digital output.*/
  185   void GPIO_SetupPinOptions(Uint16 pin, Uint16 output, Uint16 flags)
  186   {
  187       {
  188       volatile Uint32 *gpioBaseAddr;
  189       volatile Uint32 *dir, *pud, *qsel;
  190       Uint32 pin32, pin16, pinMask, qual;
  191       
  192       pin32 = pin % 32;
  193       pin16 = pin % 16;
  194       pinMask = 1UL << pin32;
  195       #ifdef MW_PIL_TIC2000_CONCERTO
  196       if (pin >= 128)
  197       {
  198           #ifdef F28M36x_DEVICE_H
  199           if (pin <= 135)
  200           {
  201               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;
  202           }
  203           else if (pin >=192 && pin <= 199)
  204           {
  205               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x1080/2);
  206           }
  207           #endif //End of F28M36x_DEVICE_H
  208           #ifdef F28M35x_DEVICE_H
  209           if (pin <= 135)
  210           {
  211               gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x1000/2);
  212           }
  213           #endif //End of F28M35x_DEVICE_H
  214       }
  215       else
  216       {
  217           #ifdef F28M36x_DEVICE_H
  218           gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (0x80/2) + (pin/32)*GPY_CTRL_OFFSET;
  219           #endif //End of F28M36x_DEVICE_H
  220           #ifdef F28M35x_DEVICE_H
  221           gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;
  222           #endif //End of F28M35x_DEVICE_H
  223       }
  224       #else
  225       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (pin/32)*GPY_CTRL_OFFSET;
  226       #endif // End of MW_PIL_TIC2000_CONCERTO
  227       
  228       /*Create pointers to the appropriate registers. This is a workaround
  229        * for the way GPIO registers are defined. The standard definition
  230        * in the header file makes it very easy to do named accesses of one
  231        * register or bit, but hard to do arbitrary numerical accesses. It's
  232        * easier to have an array of GPIO modules with identical registers,
  233        * including arrays for multi-register groups like GPyQSEL1-2. But
  234        * the header file doesn't define anything we can turn into an array,
  235        * so manual pointer arithmetic is used instead.*/
  236       dir = gpioBaseAddr + GPYDIR;
  237       pud = gpioBaseAddr + GPYPUD;
  238       qsel = gpioBaseAddr + GPYQSEL + pin32/16;
  239       
  240       EALLOW;
  241       
  242       /*Set the data direction*/
  243       *dir &= ~pinMask;
  244       if (output == 1)
  245       {
  246           /*Output, with optional open drain mode and pull-up*/
  247           *dir |= pinMask;
  248           
  249           /*Enable pull-up if necessary. Open drain mode must be active.*/
  250           if (flags & GPIO_PULLUP)
  251               *pud &= ~pinMask;
  252           else
  253               *pud |= pinMask;
  254       } else
  255       {
  256           /*Input, with optional pull-up, qualification, and polarity inversion*/
  257           *dir &= ~pinMask;
  258           
  259           /*Enable pull-up if necessary*/
  260           if (flags & GPIO_PULLUP)
  261               *pud &= ~pinMask;
  262           else
  263               *pud |= pinMask;
  264       }
  265       
  266       /*Extract the qualification parameter and load it into the register. This is
  267        * also needed for open drain outputs, so we might as well do it all the time.*/
  268       qual = (flags & GPIO_ASYNC) / GPIO_QUAL3;
  269       *qsel &= ~(0x3L << (2 * pin16));
  270       if (qual != 0x0)
  271           *qsel |= qual << (2 * pin16);
  272       
  273       EDIS;
  274       }
  275   }

  277   /*Read the GPyDAT register bit for the specified pin.
  278    * Note that this returns the actual state of the pin, not the state of the output latch.*/
  279   Uint16 GPIO_ReadPin(Uint16 pin)
  280   {
  281       volatile Uint32 *gpioDataReg;
  282       Uint16 pinVal;
  283   
  284       #ifdef MW_PIL_TIC2000_CONCERTO
  285       if (pin >= 128)
  286       {
  287           #ifdef F28M36x_DEVICE_H
  288           if (pin <= 135)
  289           {
  290               gpioDataReg = (Uint32 *)&GpioDataRegs + (0x40/2);
  291           }
  292           else if (pin >=192 && pin <= 199)
  293           {
  294               gpioDataReg = (Uint32 *)&GpioDataRegs + (0x10C0/2);
  295           }
  296           #endif //End of F28M36x_DEVICE_H
  297           #ifdef F28M35x_DEVICE_H
  298           if (pin <= 135)
  299           {
  300               gpioDataReg = (Uint32 *)&GpioDataRegs + (0x1000/2);
  301           }
  302           #endif //End of F28M35x_DEVICE_H
  303       }
  304       else
  305       {
  306           #ifdef F28M36x_DEVICE_H
  307           gpioDataReg = (Uint32 *)&GpioDataRegs + (0xC0/2) + (pin/32)*GPY_DATA_OFFSET;
  308           #endif //End of F28M36x_DEVICE_H
  309           #ifdef F28M35x_DEVICE_H
  310           gpioDataReg = (Uint32 *)&GpioDataRegs + (pin/32)*GPY_DATA_OFFSET;
  311           #endif //End of F28M35x_DEVICE_H
  312       }
  313       #else
  314       gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (pin/32)*GPY_DATA_OFFSET;
  315       #endif // End of MW_PIL_TIC2000_CONCERTO
  316       pinVal = (gpioDataReg[GPYDAT] >> (pin % 32)) & 0x1;
  317       
  318       return pinVal;
  319   }

  416   void InitGpio()
  417   {
  418       volatile Uint32 *gpioBaseAddr;
  419       Uint16 regOffset;
  420   
  421       /*Disable pin locks*/
  422       EALLOW;
  423       GpioCtrlRegs.GPALOCK.all = 0x00000000;
  424       GpioCtrlRegs.GPBLOCK.all = 0x00000000;
  425       GpioCtrlRegs.GPCLOCK.all = 0x00000000;
  426       GpioCtrlRegs.GPDLOCK.all = 0x00000000;
  427       GpioCtrlRegs.GPELOCK.all = 0x00000000;
  428       GpioCtrlRegs.GPFLOCK.all = 0x00000000;
  429   
  430       /* Fill all registers with zeros. Writing to each register separately
  431        * for six GPIO modules would make this function *very* long.
  432        * Fortunately, we'd be writing them all with zeros anyway, so this
  433        * saves a lot of space.*/
  434       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs;
  435       for (regOffset = 0; regOffset < sizeof(GpioCtrlRegs)/2; regOffset++)
  436       {
  437           /* Hack to avoid enabling pull-ups on all pins. GPyPUD is offset
  438            * 0x0C in each register group of 0x40 words. Since this is a
  439            * 32-bit pointer, the addresses must be divided by 2.*/
  440           if (regOffset % (0x40/2) != (0x0C/2))
  441           {
  442               gpioBaseAddr[regOffset] = 0x00000000;
  443           }
  444       }
  445   
  446       gpioBaseAddr = (Uint32 *)&GpioDataRegs;
  447       for (regOffset = 0; regOffset < sizeof(GpioDataRegs)/2; regOffset++)
  448       {
  449           gpioBaseAddr[regOffset] = 0x00000000;
  450       }
  451   
  452       EDIS;
  453   }

  455   void GPIO_SetupPinMux(Uint16 gpioNumber, Uint16 cpu, Uint16 muxPosition)
  456   {
  457       volatile Uint32 *gpioBaseAddr;
  458       volatile Uint32 *mux, *gmux, *csel;
  459       Uint16 pin32, pin16, pin8;
  460   
  461       pin32 = gpioNumber % 32;
  462       pin16 = gpioNumber % 16;
  463       pin8 = gpioNumber % 8;
  464       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (gpioNumber/32)*GPY_CTRL_OFFSET;
  465   
  466       /*Sanity check for valid cpu and peripheral values*/
  467       if (cpu > GPIO_MUX_CM || muxPosition > 0xF)
  468           return;
  469   
  470       /* Create pointers to the appropriate registers. This is a workaround
  471        * for the way GPIO registers are defined. The standard definition
  472        * in the header file makes it very easy to do named accesses of one
  473        * register or bit, but hard to do arbitrary numerical accesses. It's
  474        * easier to have an array of GPIO modules with identical registers,
  475        * including arrays for multi-register groups like GPyCSEL1-4. But
  476        * the header file doesn't define anything we can turn into an array,
  477        * so manual pointer arithmetic is used instead.*/
  478       mux = gpioBaseAddr + GPYMUX + pin32/16;
  479       gmux = gpioBaseAddr + GPYGMUX + pin32/16;
  480       csel = gpioBaseAddr + GPYCSEL + pin32/8;
  481   
  482       /* Now for the actual function */
  483       EALLOW;
  484   
  485       /* To change the muxing, set the peripheral mux to 0/GPIO first to avoid
  486        * glitches, then change the group mux, then set the peripheral mux to
  487        * its target value. Finally, set the CPU select. This procedure is
  488        * described in the TRM. Unfortunately, since we don't know the pin in
  489        * advance we can't hardcode a bitfield reference, so there's some
  490        * tricky bit twiddling here.*/
  491       *mux &= ~(0x3UL << (2*pin16));
  492       *gmux &= ~(0x3UL << (2*pin16));
  493       *gmux |= (Uint32)((muxPosition >> 2) & 0x3UL) << (2*pin16);
  494       *mux |= (Uint32)(muxPosition & 0x3UL) << (2*pin16);
  495   
  496       *csel &= ~(0x7L << (4*pin8));
  497       *csel |= (Uint32)(cpu & 0x7L) << (4*pin8);
  498   
  499       /* WARNING: This code does not touch the analog mode select registers,
  500        * which are needed to give the USB module control of its IOs.*/
  501       EDIS;
  502   }

  504   void GPIO_SetupPinOptions(Uint16 gpioNumber, Uint16 output, Uint16 flags)
  505   {
  506       volatile Uint32 *gpioBaseAddr;
  507       volatile Uint32 *dir, *pud, *inv, *odr, *qsel;
  508       Uint32 pin32, pin16, pinMask, qual;
  509   
  510       pin32 = gpioNumber % 32;
  511       pin16 = gpioNumber % 16;
  512       pinMask = 1UL << pin32;
  513       gpioBaseAddr = (Uint32 *)&GpioCtrlRegs + (gpioNumber/32)*GPY_CTRL_OFFSET;
  514   
  515       /* Create pointers to the appropriate registers. This is a workaround
  516        * for the way GPIO registers are defined. The standard definition
  517        * in the header file makes it very easy to do named accesses of one
  518        * register or bit, but hard to do arbitrary numerical accesses. It's
  519        * easier to have an array of GPIO modules with identical registers,
  520        * including arrays for multi-register groups like GPyQSEL1-4. But
  521        * the header file doesn't define anything we can turn into an array,
  522        * so manual pointer arithmetic is used instead.*/
  523       dir = gpioBaseAddr + GPYDIR;
  524       pud = gpioBaseAddr + GPYPUD;
  525       inv = gpioBaseAddr + GPYINV;
  526       odr = gpioBaseAddr + GPYODR;
  527       qsel = gpioBaseAddr + GPYQSEL + pin32/16;
  528   
  529       EALLOW;
  530   
  531       /*Set the data direction*/
  532       *dir &= ~pinMask;
  533       if (output == 1)
  534       {        
  535           /* Output, with optional open drain mode and pull-up */        
  536           *dir |= pinMask;
  537   
  538           /* Enable open drain if necessary */
  539           if (flags & GPIO_OPENDRAIN)
  540           {
  541               *odr |= pinMask;
  542           }
  543           else
  544           {
  545               *odr &= ~pinMask;
  546           }
  547   
  548           /* Enable pull-up if necessary. Open drain mode must be active. */
  549           if (flags & (GPIO_OPENDRAIN | GPIO_PULLUP))
  550           {
  551               *pud &= ~pinMask;
  552           }
  553           else
  554           {
  555               *pud |= pinMask;
  556           }
  557       }
  558       else
  559       {
  560           /* Input, with optional pull-up, qualification, and polarity
  561            * inversion */
  562           *dir &= ~pinMask;
  563   
  564           /* Enable pull-up if necessary */
  565           if (flags & GPIO_PULLUP)
  566           {
  567               *pud &= ~pinMask;
  568           }
  569           else
  570           {
  571               *pud |= pinMask;
  572           }
  573   
  574           /* Invert polarity if necessary */
  575           if (flags & GPIO_INVERT)
  576           {
  577               *inv |= pinMask;
  578           }
  579           else
  580           {
  581               *inv &= ~pinMask;
  582           }
  583       }
  584   
  585       /* Extract the qualification parameter and load it into the register.
  586        * This is also needed for open drain outputs, so we might as well do it
  587        * all the time. */
  588       qual = (flags & GPIO_ASYNC) / GPIO_QUAL3;
  589       *qsel &= ~(0x3L << (2 * pin16));
  590       if (qual != 0x0)
  591       {
  592           *qsel |= qual << (2 * pin16);
  593       }
  594   
  595       EDIS;
  596   }

  642   /*Read the GPyDAT register bit for the specified pin.
  643    * Note that this returns the actual state of the pin, not the state of the output latch.*/
  644   Uint16 GPIO_ReadPin(Uint16 pin)
  645   {
  646       volatile Uint32 *gpioDataReg;
  647       Uint16 pinVal;
  648   
  649       gpioDataReg = (volatile Uint32 *)&GpioDataRegs + (pin/32)*GPY_DATA_OFFSET;
  650       pinVal = (gpioDataReg[GPYDAT] >> (pin % 32)) & 0x1;
  651       
  652       return pinVal;
  653   }

  688   /*Set the GPIO pin. Set the mux values and pin options for the specified pin.
  689     This function can be used for all processors except C281x. The implementation is dependant on CPU1 and CPU2*/ 
  690   void Set_GPIOPin(Uint16 pin, Uint16 cpu, Uint16 peripheral, Uint16 output, Uint16 flags)
  691   {
  692   #if defined(F2837X_REG_FORMAT) && defined(CPU2)
  693       Uint32 *pulMsgRam = (void *)CPU01_TO_CPU02_PASSMSG;
  694       Uint32 gpioData = 0x00000000;
  695       //Sanity check for valid cpu and peripheral values
  696       if (cpu > GPIO_MUX_CPU2CLA || peripheral > 0xF)
  697           return;
  698       gpioData = gpioData | pin;
  699       gpioData = gpioData | ((Uint32)peripheral << 8);
  700       gpioData = gpioData | ((Uint32)output << 16);
  701       gpioData = gpioData | ((Uint32)flags << 24);
  702       #ifdef MW_F2837XD
  703       Uint32 ulRWord32 = 0;
  704       IPCLiteLtoRFunctionCall(IPC_FLAG0, pulMsgRam[0], gpioData, IPC_FLAG31);
  705       while (IPCLiteLtoRGetResult(&ulRWord32,IPC_LENGTH_32_BITS,
  706               IPC_FLAG31) != STATUS_PASS)
  707       {
  708       }
  709       #else
  710        MW_IPC_sendCommand(IPC_CPU2_L_CPU1_R, (IPC_FLAG0 | IPC_FLAG31), IPC_ADDR_CORRECTION_DISABLE, IPC_FUNC_CALL, pulMsgRam[0], gpioData);
  711       IPC_waitForAck(IPC_CPU2_L_CPU1_R, IPC_FLAG31);
  712       IPC_getResponse(IPC_CPU2_L_CPU1_R);
  713       #endif //MW_F2837XD
  714   #else
  715       GPIO_SetupPinMux(pin, cpu, peripheral);
  716       GPIO_SetupPinOptions(pin, output, flags);
  717   #endif // end of defined(F2837X_REG_FORMAT) && defined(CPU2)
  718   }
