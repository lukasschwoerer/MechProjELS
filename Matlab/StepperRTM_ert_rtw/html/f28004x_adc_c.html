   49   //
   50   // SetVREF - Set Vref mode. Function to select reference mode and offset trim.
   51   // Offset trim for Internal VREF 3.3 is unique.  All other modes use the same
   52   // offset trim. Also note that when the mode parameter is ADC_EXTERNAL, the
   53   // ref parameter has no effect.
   54   //
   55   void SetVREF(int module, int mode, int ref)
   56   {
   57       Uint16 *offset, offval;
   58   
   59       //
   60       // Define offset locations from OTP
   61       //
   62       offset = (Uint16 *)(0x70594 + (module * 6));
   63   
   64       if((mode == ADC_INTERNAL) && (ref == ADC_VREF3P3))
   65       {
   66           offval = (*offset) >> 8;    // Internal / 1.65v mode offset
   67       }
   68       else
   69       {
   70           offval = (*offset) & 0xFF;  // All other modes
   71       }
   72   
   73       //
   74       // Write offset trim values and configure reference modes
   75       //
   76       EALLOW;
   77       switch(module)
   78       {
   79           case 0:
   80               AdcaRegs.ADCOFFTRIM.bit.OFFTRIM = offval;
   81               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFASEL = mode;
   82               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFA2P5SEL = ref;
   83               break;
   84           case 1:
   85               AdcbRegs.ADCOFFTRIM.bit.OFFTRIM = offval;
   86               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFBSEL = mode;
   87               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFB2P5SEL = ref;
   88               break;
   89           case 2:
   90               AdccRegs.ADCOFFTRIM.bit.OFFTRIM = offval;
   91               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFCSEL = mode;
   92               AnalogSubsysRegs.ANAREFCTL.bit.ANAREFC2P5SEL = ref;
   93               break;
   94           default:
   95               break;
   96       }
   97       EDIS;
   98   }
