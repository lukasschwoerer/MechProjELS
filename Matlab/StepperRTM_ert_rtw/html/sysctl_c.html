   76   //
   77   // SysCtl_pollX1Counter()
   78   //
   79   //*****************************************************************************
   80   static void
   81   SysCtl_pollX1Counter(void)
   82   {
   83       uint16_t loopCount = 0U;
   84   
   85       //
   86       // Delay for 1 ms while the XTAL powers up
   87       //
   88       // 2000 loops, 5 cycles per loop + 9 cycles overhead = 10009 cycles
   89       //
   90       SysCtl_delay(2000);
   91   
   92       //
   93       // Clear and saturate X1CNT 4 times to guarantee operation
   94       //
   95       do
   96       {
   97           //
   98           // Keep clearing the counter until it is no longer saturated
   99           //
  100           while(HWREG(CLKCFG_BASE + SYSCTL_O_X1CNT) > 0x1FFU)
  101           {
  102               HWREG(CLKCFG_BASE + SYSCTL_O_X1CNT) |= SYSCTL_X1CNT_CLR;
  103           }
  104   
  105           //
  106           // Wait for the X1 clock to saturate
  107           //
  108           while(HWREGH(CLKCFG_BASE + SYSCTL_O_X1CNT) != SYSCTL_X1CNT_X1CNT_M)
  109           {
  110               //
  111               // If your application is stuck in this loop, please check if the
  112               // input clock source is valid.
  113               //
  114           }
  115   
  116           //
  117           // Increment the counter
  118           //
  119           loopCount++;
  120       }while(loopCount < 4U);
  121   }

  123   //*****************************************************************************
  124   //
  125   // SysCtl_getClock()
  126   //
  127   //*****************************************************************************
  128   uint32_t
  129   SysCtl_getClock(uint32_t clockInHz)
  130   {
  131       uint32_t temp;
  132       uint32_t oscSource;
  133       uint32_t clockOut;
  134   
  135       //
  136       // Don't proceed if an MCD failure is detected.
  137       //
  138       if(SysCtl_isMCDClockFailureDetected())
  139       {
  140           //
  141           // OSCCLKSRC2 failure detected. Returning the INTOSC1 rate. You need
  142           // to handle the MCD and clear the failure.
  143           //
  144           clockOut = SYSCTL_DEFAULT_OSC_FREQ;
  145       }
  146       else
  147       {
  148           //
  149           // If one of the internal oscillators is being used, start from the
  150           // known default frequency.  Otherwise, use clockInHz parameter.
  151           //
  152           oscSource = HWREG(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &
  153                       (uint32_t)SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M;
  154   
  155           if((oscSource == ((uint32_t)SYSCTL_OSCSRC_OSC2 >> SYSCTL_OSCSRC_S)) ||
  156              (oscSource == ((uint32_t)SYSCTL_OSCSRC_OSC1 >> SYSCTL_OSCSRC_S)))
  157           {
  158               clockOut = SYSCTL_DEFAULT_OSC_FREQ;
  159           }
  160           else
  161           {
  162               clockOut = clockInHz;
  163           }
  164   
  165           //
  166           // If the PLL is enabled calculate its effect on the clock
  167           //
  168           if((HWREG(CLKCFG_BASE + SYSCTL_O_SYSPLLCTL1) &
  169               (SYSCTL_SYSPLLCTL1_PLLEN | SYSCTL_SYSPLLCTL1_PLLCLKEN)) == 3U)
  170           {
  171               //
  172               // Calculate portion from fractional multiplier
  173               //
  174               temp = (clockInHz * ((HWREG(CLKCFG_BASE + SYSCTL_O_SYSPLLMULT) &
  175                                     SYSCTL_SYSPLLMULT_FMULT_M) >>
  176                                    SYSCTL_SYSPLLMULT_FMULT_S)) / 4U;
  177   
  178               //
  179               // Calculate integer multiplier and fixed divide by 2
  180               //
  181               clockOut = clockOut * ((HWREG(CLKCFG_BASE + SYSCTL_O_SYSPLLMULT) &
  182                                       SYSCTL_SYSPLLMULT_IMULT_M) >>
  183                                      SYSCTL_SYSPLLMULT_IMULT_S);
  184   
  185               //
  186               // Add in fractional portion
  187               //
  188               clockOut += temp;
  189           }
  190   
  191           if((HWREG(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) &
  192               SYSCTL_SYSCLKDIVSEL_PLLSYSCLKDIV_M) != 0U)
  193           {
  194               clockOut /= (2U * (HWREG(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) &
  195                                  SYSCTL_SYSCLKDIVSEL_PLLSYSCLKDIV_M));
  196           }
  197       }
  198   
  199       return(clockOut);
  200   }

  202   //*****************************************************************************
  203   //
  204   // SysCtl_setClock()
  205   //
  206   //*****************************************************************************
  207   bool
  208   SysCtl_setClock(uint32_t config)
  209   {
  210       uint16_t divSel;
  211       uint16_t pllMult;
  212       uint32_t retries, oscSource, pllLockStatus;
  213       uint32_t timeout;
  214       bool status = false;
  215       uint16_t mult;
  216   
  217       //
  218       // Check the arguments.
  219       //
  220       ASSERT((config & SYSCTL_OSCSRC_M) <= SYSCTL_OSCSRC_M);
  221   
  222       //
  223       // Don't proceed to the PLL initialization if an MCD failure is detected.
  224       //
  225       if(SysCtl_isMCDClockFailureDetected())
  226       {
  227           //
  228           // OSCCLKSRC2 failure detected. Returning false. You'll need to clear
  229           // the MCD error.
  230           //
  231           status = false;
  232       }
  233       else
  234       {
  235           //
  236           // Configure oscillator source
  237           //
  238           oscSource = config & SYSCTL_OSCSRC_M;
  239           SysCtl_selectOscSource(oscSource);
  240   
  241           //
  242           // Bypass PLL
  243           //
  244           EALLOW;
  245           HWREGH(CLKCFG_BASE + SYSCTL_O_SYSPLLCTL1) &=
  246               ~SYSCTL_SYSPLLCTL1_PLLCLKEN;
  247           EDIS;
  248   
  249           //
  250           // Delay of at least 60 OSCCLK cycles required post PLL bypass
  251           //
  252           SysCtl_delay(11U);
  253   
  254           //
  255           // Get the PLL multiplier settings from config
  256           //
  257           pllMult  = (uint16_t)((config & SYSCTL_IMULT_M) <<
  258                                 SYSCTL_SYSPLLMULT_IMULT_S);
  259   
  260           pllMult |= (uint16_t)(((config & SYSCTL_FMULT_M) >>
  261                                  SYSCTL_FMULT_S) <<
  262                                 SYSCTL_SYSPLLMULT_FMULT_S);
  263   
  264           //
  265           // Get the PLL multipliers currently programmed
  266           //
  267           mult  = (uint16_t)((HWREG(CLKCFG_BASE + SYSCTL_O_SYSPLLMULT) &
  268                               (uint32_t)SYSCTL_SYSPLLMULT_IMULT_M) >>
  269                              (uint32_t)SYSCTL_SYSPLLMULT_IMULT_S);
  270   
  271           mult |= (uint16_t)(HWREG(CLKCFG_BASE + SYSCTL_O_SYSPLLMULT) &
  272                                    SYSCTL_SYSPLLMULT_FMULT_M);
  273   
  274           //
  275           // Lock PLL only if the multipliers need update
  276           //
  277           if(mult !=  pllMult)
  278           {
  279               //
  280               // Configure PLL if enabled
  281               //
  282               if((config & SYSCTL_PLL_ENABLE) == SYSCTL_PLL_ENABLE)
  283               {
  284                   //
  285                   // Set dividers to /1
  286                   //
  287                   EALLOW;
  288                   HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) = 0U;
  289                   EDIS;
  290   
  291                   //
  292                   // Loop to retry locking the PLL should the DCC module
  293                   // indicate that it was not successful.
  294                   //
  295                   for(retries = 0U; (retries < SYSCTL_PLL_RETRIES); retries++)
  296                   {
  297                       //
  298                       // Turn off PLL
  299                       //
  300                       EALLOW;
  301                       HWREGH(CLKCFG_BASE + SYSCTL_O_SYSPLLCTL1) &=
  302                           ~SYSCTL_SYSPLLCTL1_PLLEN;
  303   
  304                       SysCtl_delay(3U);
  305   
  306                       //
  307                       // Write multiplier, which automatically turns on the PLL
  308                       //
  309                       HWREGH(CLKCFG_BASE + SYSCTL_O_SYSPLLMULT) = pllMult;
  310   
  311                       //
  312                       // Wait for the SYSPLL lock counter or a timeout
  313                       //
  314                       timeout = SYSCTL_PLLLOCK_TIMEOUT;
  315                       pllLockStatus = HWREGH(CLKCFG_BASE + SYSCTL_O_SYSPLLSTS) &
  316                                       SYSCTL_SYSPLLSTS_LOCKS;
  317   
  318                       while((pllLockStatus != 1U) && (timeout != 0U))
  319                       {
  320                           pllLockStatus = HWREGH(CLKCFG_BASE +
  321                                                  SYSCTL_O_SYSPLLSTS) &
  322                                           SYSCTL_SYSPLLSTS_LOCKS;
  323                           timeout--;
  324                       }
  325                       EDIS;
  326   
  327                       //
  328                       // Check PLL Frequency using DCC
  329                       //
  330                       status = SysCtl_isPLLValid(oscSource,
  331                                                 (config &
  332                                                 ((uint32_t)SYSCTL_IMULT_M |
  333                                                  (uint32_t)SYSCTL_FMULT_M)));
  334   
  335                       //
  336                       // Check DCC Status, if no error break the loop
  337                       //
  338                       if(status)
  339                       {
  340                           break;
  341                       }
  342                   }
  343               }
  344               else
  345               {
  346                   status = true;
  347               }
  348           }
  349         else
  350           {
  351               status = true;
  352           }
  353   
  354   
  355           //
  356           // If PLL locked successfully, configure the dividers
  357           //
  358           if(status)
  359           {
  360               //
  361               // Set divider to produce slower output frequency to limit current
  362               // increase.
  363               //
  364               divSel = (uint16_t)(config & SYSCTL_SYSDIV_M) >> SYSCTL_SYSDIV_S;
  365   
  366               EALLOW;
  367               if(divSel != (126U / 2U))
  368               {
  369                   HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) =
  370                       (HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) &
  371                        ~(uint16_t)SYSCTL_SYSCLKDIVSEL_PLLSYSCLKDIV_M) |
  372                       (divSel + 1U);
  373               }
  374               else
  375               {
  376                   HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) =
  377                       (HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) &
  378                        ~(uint16_t)SYSCTL_SYSCLKDIVSEL_PLLSYSCLKDIV_M) | divSel;
  379               }
  380   
  381               EDIS;
  382   
  383               //
  384               // Enable PLLSYSCLK is fed from system PLL clock
  385               //
  386               EALLOW;
  387               HWREGH(CLKCFG_BASE + SYSCTL_O_SYSPLLCTL1) |=
  388                   SYSCTL_SYSPLLCTL1_PLLCLKEN;
  389               EDIS;
  390   
  391               //
  392               // ~200 PLLSYSCLK delay to allow voltage regulator to stabilize
  393               // prior to increasing entire system clock frequency.
  394               //
  395               SysCtl_delay(40U);
  396   
  397               //
  398               // Set the divider to user value
  399               //
  400               EALLOW;
  401               HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) =
  402                   (HWREGH(CLKCFG_BASE + SYSCTL_O_SYSCLKDIVSEL) &
  403                    ~SYSCTL_SYSCLKDIVSEL_PLLSYSCLKDIV_M) | divSel;
  404               EDIS;
  405           }
  406       }
  407   
  408       return(status);
  409   }
  410   //*****************************************************************************
  411   //
  412   // SysCtl_selectXTAL()
  413   //
  414   //*****************************************************************************
  415   void
  416   SysCtl_selectXTAL(void)
  417   {
  418       EALLOW;
  419   
  420       //
  421       // Turn on XTAL and select crystal mode
  422       //
  423       HWREGH(CLKCFG_BASE + SYSCTL_O_XTALCR) &= ~SYSCTL_XTALCR_OSCOFF;
  424       HWREGH(CLKCFG_BASE + SYSCTL_O_XTALCR) &= ~SYSCTL_XTALCR_SE;
  425       EDIS;
  426   
  427       //
  428       // Wait for the X1 clock to saturate
  429       //
  430       SysCtl_pollX1Counter();
  431   
  432       //
  433       // Select XTAL as the oscillator source
  434       //
  435       EALLOW;
  436       HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) =
  437       ((HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &
  438         (~SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M)) |
  439        ((uint32_t)SYSCTL_OSCSRC_XTAL >> SYSCTL_OSCSRC_S));
  440       EDIS;
  441   
  442       //
  443       // If a missing clock failure was detected, try waiting for the X1 counter
  444       // to saturate again. Consider modifying this code to add a 10ms timeout.
  445       //
  446       while(SysCtl_isMCDClockFailureDetected())
  447       {
  448           //
  449           // Clear the MCD failure
  450           //
  451           SysCtl_resetMCD();
  452   
  453           //
  454           // Wait for the X1 clock to saturate
  455           //
  456           SysCtl_pollX1Counter();
  457   
  458           //
  459           // Select XTAL as the oscillator source
  460           //
  461           EALLOW;
  462           HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) =
  463           ((HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &
  464             (~SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M)) |
  465            ((uint32_t)SYSCTL_OSCSRC_XTAL >> SYSCTL_OSCSRC_S));
  466           EDIS;
  467       }
  468   }

  471   //
  472   // SysCtl_selectXTALSingleEnded()
  473   //
  474   //*****************************************************************************
  475   void
  476   SysCtl_selectXTALSingleEnded(void)
  477   {
  478       //
  479       // Turn on XTAL and select single-ended mode.
  480       //
  481       EALLOW;
  482       HWREGH(CLKCFG_BASE + SYSCTL_O_XTALCR) &= ~SYSCTL_XTALCR_OSCOFF;
  483       HWREGH(CLKCFG_BASE + SYSCTL_O_XTALCR) |= SYSCTL_XTALCR_SE;
  484       EDIS;
  485   
  486       //
  487       // Wait for the X1 clock to saturate
  488       //
  489       SysCtl_pollX1Counter();
  490   
  491       //
  492       // Select XTAL as the oscillator source
  493       //
  494       EALLOW;
  495       HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) =
  496       ((HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &
  497         (~SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M)) |
  498        ((uint32_t)SYSCTL_OSCSRC_XTAL >> SYSCTL_OSCSRC_S));
  499       EDIS;
  500   
  501       //
  502       // Something is wrong with the oscillator module. Replace the ESTOP0 with
  503       // an appropriate error-handling routine.
  504       //
  505       while(SysCtl_isMCDClockFailureDetected())
  506       {
  507           ESTOP0;
  508       }
  509   }

  511   //*****************************************************************************
  512   //
  513   // SysCtl_selectOscSource()
  514   //
  515   //*****************************************************************************
  516   void
  517   SysCtl_selectOscSource(uint32_t oscSource)
  518   {
  519       ASSERT((oscSource == SYSCTL_OSCSRC_OSC1) |
  520              (oscSource == SYSCTL_OSCSRC_OSC2) |
  521              (oscSource == SYSCTL_OSCSRC_XTAL) |
  522              (oscSource == SYSCTL_OSCSRC_XTAL_SE));
  523   
  524       //
  525       // Select the specified source.
  526       //
  527       EALLOW;
  528       switch(oscSource)
  529       {
  530           case SYSCTL_OSCSRC_OSC2:
  531               //
  532               // Turn on INTOSC2
  533               //
  534               HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &=
  535                   ~SYSCTL_CLKSRCCTL1_INTOSC2OFF;
  536   
  537               SYSCTL_CLKSRCCTL1_DELAY;
  538   
  539               //
  540               // Clk Src = INTOSC2
  541               //
  542               HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &=
  543                   ~SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M;
  544   
  545               break;
  546   
  547           case SYSCTL_OSCSRC_XTAL:
  548               //
  549               // Select XTAL in crystal mode and wait for it to power up
  550               //
  551               SysCtl_selectXTAL();
  552               break;
  553   
  554           case SYSCTL_OSCSRC_XTAL_SE:
  555               //
  556               // Select XTAL in single-ended mode and wait for it to power up
  557               //
  558               SysCtl_selectXTALSingleEnded();
  559               break;
  560   
  561           case SYSCTL_OSCSRC_OSC1:
  562               //
  563               // Clk Src = INTOSC1
  564               //
  565               HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) =
  566                      (HWREGH(CLKCFG_BASE + SYSCTL_O_CLKSRCCTL1) &
  567                       ~SYSCTL_CLKSRCCTL1_OSCCLKSRCSEL_M) |
  568                      ((uint32_t)SYSCTL_OSCSRC_OSC1 >> SYSCTL_OSCSRC_S);
  569   
  570               break;
  571   
  572           default:
  573               //
  574               // Do nothing. Not a valid oscSource value.
  575               //
  576               break;
  577       }
  578       EDIS;
  579   }

  581   //*****************************************************************************
  582   //
  583   // SysCtl_getLowSpeedClock()
  584   //
  585   //*****************************************************************************
  586   uint32_t
  587   SysCtl_getLowSpeedClock(uint32_t clockInHz)
  588   {
  589       uint32_t clockOut;
  590   
  591       //
  592       // Get the main system clock
  593       //
  594       clockOut = SysCtl_getClock(clockInHz);
  595   
  596       //
  597       // Apply the divider to the main clock
  598       //
  599       if((HWREG(CLKCFG_BASE + SYSCTL_O_LOSPCP) &
  600           SYSCTL_LOSPCP_LSPCLKDIV_M) != 0U)
  601       {
  602           clockOut /= (2U * (HWREG(CLKCFG_BASE + SYSCTL_O_LOSPCP) &
  603                               SYSCTL_LOSPCP_LSPCLKDIV_M));
  604       }
  605   
  606       return(clockOut);
  607   }

  609   //*****************************************************************************
  610   //
  611   // SysCtl_getDeviceParametric()
  612   //
  613   //*****************************************************************************
  614   uint16_t
  615   SysCtl_getDeviceParametric(SysCtl_DeviceParametric parametric)
  616   {
  617       uint32_t value;
  618   
  619       //
  620       // Get requested parametric value
  621       //
  622       switch(parametric)
  623       {
  624           case SYSCTL_DEVICE_QUAL:
  625               //
  626               // Qualification Status
  627               //
  628               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDL) &
  629                         SYSCTL_PARTIDL_QUAL_M) >> SYSCTL_PARTIDL_QUAL_S);
  630               break;
  631   
  632           case SYSCTL_DEVICE_PINCOUNT:
  633               //
  634               // Pin Count
  635               //
  636               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDL) &
  637                         SYSCTL_PARTIDL_PIN_COUNT_M) >>
  638                        SYSCTL_PARTIDL_PIN_COUNT_S);
  639               break;
  640   
  641           case SYSCTL_DEVICE_INSTASPIN:
  642               //
  643               // InstaSPIN Feature Set
  644               //
  645               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDL) &
  646                         SYSCTL_PARTIDL_INSTASPIN_M) >>
  647                        SYSCTL_PARTIDL_INSTASPIN_S);
  648               break;
  649   
  650           case SYSCTL_DEVICE_FLASH:
  651               //
  652               // Flash Size (KB)
  653               //
  654               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDL) &
  655                         SYSCTL_PARTIDL_FLASH_SIZE_M) >>
  656                        SYSCTL_PARTIDL_FLASH_SIZE_S);
  657               break;
  658           case SYSCTL_DEVICE_FAMILY:
  659               //
  660               // Device Family
  661               //
  662               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDH) &
  663                         SYSCTL_PARTIDH_FAMILY_M) >> SYSCTL_PARTIDH_FAMILY_S);
  664               break;
  665   
  666           case SYSCTL_DEVICE_PARTNO:
  667               //
  668               // Part Number
  669               //
  670               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDH) &
  671                         SYSCTL_PARTIDH_PARTNO_M) >> SYSCTL_PARTIDH_PARTNO_S);
  672               break;
  673   
  674           case SYSCTL_DEVICE_CLASSID:
  675               //
  676               // Class ID
  677               //
  678               value = ((HWREG(DEVCFG_BASE + SYSCTL_O_PARTIDH) &
  679                         SYSCTL_PARTIDH_DEVICE_CLASS_ID_M) >>
  680                        SYSCTL_PARTIDH_DEVICE_CLASS_ID_S);
  681               break;
  682   
  683           default:
  684               //
  685               // Not a valid value for PARTID register
  686               //
  687               value = 0U;
  688               break;
  689       }
  690   
  691       return((uint16_t)value);
  692   }
  693   //*****************************************************************************
  694   //
  695   // SysCtl_isPLLValid()
  696   //
  697   //*****************************************************************************
  698   bool
  699   SysCtl_isPLLValid(uint32_t oscSource, uint32_t pllMult)
  700   {
  701       uint32_t imult, fmult, base;
  702   
  703       DCC_Count0ClockSource dccClkSrc0;
  704       DCC_Count1ClockSource dccClkSrc1;
  705       uint32_t dccCounterSeed0, dccCounterSeed1, dccValidSeed0;
  706   
  707       switch(oscSource)
  708       {
  709           case SYSCTL_OSCSRC_OSC2:
  710               //
  711               // Select DCC Clk Src0 as INTOSC2
  712               //
  713               dccClkSrc0 = DCC_COUNT0SRC_INTOSC2;
  714               break;
  715           case SYSCTL_OSCSRC_XTAL:
  716           case SYSCTL_OSCSRC_XTAL_SE:
  717               //
  718               // Select DCC Clk Src0 as XTAL
  719               //
  720               dccClkSrc0 = DCC_COUNT0SRC_XTAL;
  721               break;
  722           case SYSCTL_OSCSRC_OSC1:
  723               //
  724               // Select DCC Clk Src0 as INTOSC1
  725               //
  726               dccClkSrc0 = DCC_COUNT0SRC_INTOSC1;
  727               break;
  728           default:
  729               //
  730               // Select DCC Clk Src0 as INTOSC1
  731               //
  732               dccClkSrc0 = DCC_COUNT0SRC_INTOSC1;
  733               break;
  734       }
  735   
  736       //
  737       // Setting Counter0 & Valid Seed Value with +/-2% tolerance
  738       //
  739       dccCounterSeed0 = (uint32_t)SYSCTL_DCC_COUNTER0_WINDOW - 2U;
  740       dccValidSeed0 = 4U;
  741   
  742       //
  743       // Select DCC0 for PLL validation
  744       //
  745       base = DCC0_BASE;
  746   
  747       //
  748       // Select DCC Clk Src1 as SYSPLL
  749       //
  750       dccClkSrc1 = DCC_COUNT1SRC_PLL;
  751   
  752       imult = pllMult & SYSCTL_IMULT_M;
  753       fmult = pllMult & SYSCTL_FMULT_M;
  754   
  755       //
  756       // Multiplying Counter-0 window with PLL Integer Multiplier
  757       //
  758       dccCounterSeed1 = SYSCTL_DCC_COUNTER0_WINDOW * imult;
  759   
  760       //
  761       // Multiplying Counter-0 window with PLL Fractional Multiplier
  762       //
  763       switch(fmult)
  764       {
  765           case SYSCTL_FMULT_1_4:
  766               //
  767               // FMULT * CNTR0 Window = 0.25 * 100 = 25, gets added to cntr0
  768               // seed value
  769               //
  770               dccCounterSeed1 = dccCounterSeed1 + 25U;
  771               break;
  772           case SYSCTL_FMULT_1_2:
  773               //
  774               // FMULT * CNTR0 Window = 0.5 * 100 = 50, gets added to cntr0
  775               // seed value
  776               //
  777               dccCounterSeed1 = dccCounterSeed1 + 50U;
  778               break;
  779           case SYSCTL_FMULT_3_4:
  780               //
  781               // FMULT * CNTR0 Window = 0.75 * 100 = 75, gets added to cntr0
  782               // seed value
  783               //
  784               dccCounterSeed1 = dccCounterSeed1 + 75U;
  785               break;
  786           default:
  787               //
  788               // No fractional multiplier
  789               //
  790               dccCounterSeed1 = dccCounterSeed1;
  791               break;
  792       }
  793   
  794   
  795       //
  796       // Enable peripheral clock to DCC
  797       //
  798       SysCtl_enablePeripheral(SYSCTL_PERIPH_CLK_DCC0);
  799   
  800       //
  801       // Clear Error & Done Flag
  802       //
  803       DCC_clearErrorFlag(base);
  804       DCC_clearDoneFlag(base);
  805   
  806       //
  807       // Disable DCC
  808       //
  809       DCC_disableModule(base);
  810   
  811       //
  812       // Disable Error Signal
  813       //
  814       DCC_disableErrorSignal(base);
  815   
  816       //
  817       // Disable Done Signal
  818       //
  819       DCC_disableDoneSignal(base);
  820   
  821       //
  822       // Configure Clock Source0 to whatever set as a clock source for PLL
  823       //
  824       DCC_setCounter0ClkSource(base, dccClkSrc0);
  825   
  826       //
  827       // Configure Clock Source1 to PLL
  828       //
  829       DCC_setCounter1ClkSource(base, dccClkSrc1);
  830   
  831       //
  832       // Configure COUNTER-0, COUNTER-1 & Valid Window
  833       //
  834       DCC_setCounterSeeds(base, dccCounterSeed0, dccValidSeed0,
  835                           dccCounterSeed1);
  836   
  837       //
  838       // Enable Single Shot mode
  839       //
  840       DCC_enableSingleShotMode(base, DCC_MODE_COUNTER_ZERO);
  841   
  842       //
  843       // Enable Error Signal
  844       //
  845       DCC_enableErrorSignal(base);
  846   
  847       //
  848       // Enable Done Signal
  849       //
  850       DCC_enableDoneSignal(base);
  851   
  852       //
  853       // Enable DCC to start counting
  854       //
  855       DCC_enableModule(base);
  856   
  857       //
  858       // Wait until Error or Done Flag is generated
  859       //
  860       while((HWREGH(base + DCC_O_STATUS) &
  861              (DCC_STATUS_ERR | DCC_STATUS_DONE)) == 0U)
  862       {
  863       }
  864   
  865       //
  866       // Returns true if DCC completes without error
  867       //
  868       return((HWREGH(base + DCC_O_STATUS) &
  869               (DCC_STATUS_ERR | DCC_STATUS_DONE)) == DCC_STATUS_DONE);
  870   }
