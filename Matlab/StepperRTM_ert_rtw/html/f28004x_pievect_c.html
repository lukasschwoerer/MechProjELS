  279   //
  280   // InitPieVectTable - This function initializes the PIE vector table to a known
  281   // state. This function must be executed after boot time.
  282   //
  283   void
  284   InitPieVectTable(void)
  285   {
  286       Uint16  i;
  287       Uint32  *Source  =  (void  *)  &PieVectTableInit;
  288       Uint32  *Dest  =  (void  *)  &PieVectTable;
  289   
  290       //
  291       // Do not write over first 3 32-bit locations (these locations are
  292       // initialized by Boot ROM with boot variables)
  293       //
  294       Source  =  Source  +  3;
  295       Dest  =  Dest  +  3;
  296   
  297       EALLOW;
  298   
  299       for(i  =  0;  i  <  221;  i++)
  300       {
  301           *Dest++  =  *Source++;
  302       }
  303   
  304       EDIS;
  305   
  306       //
  307       // Enable the PIE Vector Table
  308       //
  309       PieCtrlRegs.PIECTRL.bit.ENPIE  =  1;
  310   }
