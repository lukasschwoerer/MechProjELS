  131   void configureTimer0(float baseRate, float systemClock)
  132   {
  133       
  134       /* InitCpuTimers() - CPU Timers are initialized in 
  135        * MW_c28xx_board.c in the generated code.
  136        */
  137   
  138   	/* Configure CPU-Timer 0 to interrupt every base rate. */
  139   	/* Parameters:  Timer Pointer, CPU Freq in MHz, Period in usec. */
  140   	ConfigCpuTimer(&CpuTimer0, systemClock, baseRate * 1000000);
  141   	StartCpuTimer0();
  142   
  143   	EALLOW;
  144   	PieVectTable.TIMER0_INT = &TINT0_isr;     /* Hook interrupt to the ISR*/
  145   	EDIS;
  146   
  147   	PieCtrlRegs.PIEIER1.bit.INTx7 = 1;   /* Enable interrupt TINT0 */
  148   	IER |= M_INT1;
  149   }

  151   void enableTimer0Interrupt(void)
  152   {
  153   	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */
  154   	PieCtrlRegs.PIEIER1.all |= (1 << 6); /* Enable PIER bit for timer 0 */
  155   	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */
  156   }
  157   
  158   bool disableTimer0Interrupt(void)
  159   {
  160       bool isrEnabled = false;
  161   	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */
  162       isrEnabled = PieCtrlRegs.PIEIER1.all & (1 << 6) ;
  163   	PieCtrlRegs.PIEIER1.all &= ~(1 << 6); /* Disable PIER bit for timer 0 */
  164   	asm(" RPT #5 || NOP");
  165   	IFR &= 0xFFFE;
  166   	PieCtrlRegs.PIEACK.all = 0x1;
  167   	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */
  168       return isrEnabled;
  169   }
